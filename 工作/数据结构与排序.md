```c
// 单链表节点定义
typedef struct ListNode {
    int val;
    struct ListNode *next;
} ListNode;

// 双链表节点定义
typedef struct DNode {
    int data;
    struct DNode *prev;  // 前驱 
    struct DNode *next;  // 后继
} DNode;

// 创建新节点
ListNode* createNode(int val) {
    ListNode *node = (ListNode*)malloc(sizeof(ListNode));
    if (!node) {
        printf("内存分配失败\n");
        exit(1);
    }
    node->val = val;
    node->next = NULL;
    return node;
}

//  遍历打印链表
void printList(ListNode *head) {
    if (head == NULL) {
        printf("链表为空\n");
        return;
    }
    ListNode *cur = head;
    while (cur) {
        printf("%d -> ", cur->val);
        cur = cur->next;
    }
    printf("NULL\n");
}

// 尾插法
ListNode* insertAtTail(ListNode *head, int val) {
    ListNode *newNode = createNode(val);
    if (head == NULL) {
        return newNode;
    }
    ListNode *cur = head;
    while (cur->next) {
        cur = cur->next;    //遍历链表到最后一个节点
    }
    cur->next = newNode;
    return head;
}

//  删除指定位置的节点（从0开始）
ListNode* deleteAtPos(ListNode *head, int pos) {
    if (head == NULL || pos < 0) return head;

    ListNode *temp = head;
    // 删除头节点
    if (pos == 0) {
        head = head->next;
        free(temp);
        return head;
    }

    // 找到待删除节点的前一个节点
    ListNode *cur = head;
    int i = 0;
    while (cur && i < pos - 1) {
        cur = cur->next;
        i++;
    }
    // 位置越界
    if (cur == NULL || cur->next == NULL) {
        printf("删除位置非法\n");
        return head;
    }

    // 删除节点
    temp = cur->next;
    cur->next = temp->next;
    free(temp);
    return head;
}

// 反转链表
ListNode* reverseList(ListNode* head) {
    ListNode *prev = NULL, *cur = head, *next;
    while(cur) {
        next = cur->next;
        cur->next = prev;
        prev = cur;
        cur = next;
    }
    return prev;
}

// 判断链表是否有环
int hasCycle(ListNode *head) {
    ListNode *slow = head, *fast = head;
    while(fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
        if(slow == fast) return 1; // 有环
    }
    return 0; // 无环
}

// 找中间节点
ListNode* middleNode(ListNode* head) {
    ListNode *slow = head, *fast = head;
    while(fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
    }
    return slow;
}

// 合并两个有序链表
ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
    // 建一个虚拟头节点，方便操作
    ListNode dummy;
    ListNode* tail = &dummy;
    dummy.next = NULL;

    // 两个链表都不为空，比较大小，小的先接上去
    while (l1 && l2) {
        if (l1->val < l2->val) {
            tail->next = l1;
            l1 = l1->next;
        } else {
            tail->next = l2;
            l2 = l2->next;
        }
        tail = tail->next;
    }

    // 把剩下的链表直接接上
    if (l1) tail->next = l1;
    if (l2) tail->next = l2;

    return dummy.next;
}
```

## 栈(数组实现)

```c
#define MAXSIZE 100

typedef struct {
    int data[MAXSIZE];
    int top;  // 栈顶下标
} Stack;

// 初始化
void initStack(Stack *s) {
    s->top = -1;
}

// 入栈
int push(Stack *s, int x) {
    if (s->top == MAXSIZE-1) return 0;
    s->data[++s->top] = x;
    return 1;
}

// 出栈
int pop(Stack *s) {
    if (s->top == -1) return -1;
    return s->data[s->top--];
}

// 取栈顶
int top(Stack *s) {
    if (s->top == -1) return -1;
    return s->data[s->top];
}

// 括号匹配
#include <string.h>

bool isValid(char *s) {
    char stack[1000];
    int top = -1;
    int len = strlen(s);

    for (int i = 0; i < len; i++) {
        if (s[i] == '(') stack[++top] = ')';
        else if (s[i] == '[') stack[++top] = ']';
        else if (s[i] == '{') stack[++top] = '}';
        else {
            if (top == -1 || stack[top] != s[i])
                return false;
            top--;
        }
    }
    return top == -1;
}
```

## 队列(数组实现)

```c
#define MAXSIZE 100

typedef struct {
    int data[MAXSIZE];
    int front, rear;
} Queue;

// 初始化
void initQueue(Queue *q) {
    q->front = q->rear = 0;
}

// 入队
int enqueue(Queue *q, int x) {
    if ((q->rear+1)%MAXSIZE == q->front) return 0;
    q->data[q->rear] = x;
    q->rear = (q->rear+1)%MAXSIZE;
    return 1;
}

// 出队
int dequeue(Queue *q) {
    if (q->front == q->rear) return -1;
    int x = q->data[q->front];
    q->front = (q->front+1)%MAXSIZE;
    return x;
}

```

```c
    if(!r)return;

void post(TreeNode* r){
    if(!r)return;
    post(r->left);post(r->right);
    printf("%d ",r->val);
}

// 5. 求深度
int maxDepth(TreeNode* r){
    if(!r)return 0;
    int l=maxDepth(r->l),r=maxDepth(r->r);
    return l>r?l+1:r+1;
}

// 6. 翻转二叉树
TreeNode* invertTree(TreeNode* r){
    if(!r)return NULL;
    TreeNode* t=r->left;r->l=r->right;r->right=t;
    invertTree(r->left);invertTree(r->right);
    return r;
}

// 7. 判断对称
bool isSym(TreeNode* l,TreeNode* r){
    if(!l&&!r)return 1;
    if(!l||!r)return 0;
    return l->val==r->val&&isSym(l->l,r->r)&&isSym(l->r,r->l);
}
```

# 排序算法

## 冒泡排序
```c
//时间复杂度 O(n^2)
//每一轮将最大的数沉底
void bubbleSort(int arr[], int n) {
    //n个元素需要n-1轮
    for (int i = 0; i < n - 1; i++) {
        //n-i-1：因为每轮结束后，最后i个元素已经有序，无需再比较
        for (int j = 0; j < n - i - 1; j++) {
            // 比较相邻两个元素
            if (arr[j] > arr[j + 1]) {
                //大的往后冒
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}
```
## 选择排序
```c
//时间复杂度 O(n^2)
//每一轮选择最矮的一个学生站到当前队列的最前位置
void selectSort(int arr[], int n) {
    //n个元素需要n-1轮
    for (int i = 0; i < n - 1; i++) {
        //假设当前轮次第一个元素是最小
        int min = i;
        //未排序部分找到真正的最小值
        for (int j = i + 1; j < n; j++) {
            //找到更小的，更新最小值位置
            if (arr[j] < arr[min])
                min = j;
        }
        //找到的最小值与当前轮次第一个元素交换
        int temp = arr[i];
        arr[i] = arr[min];
        arr[min] = temp;
    }
}
```
## 插入排序
```c
//时间复杂度 O(n^2)
//打扑克牌理牌，左边的部分已经排序，右边的部分还没排序，每次摸一张牌，都在正确的位置插入
void insertSort(int arr[], int n) {
    int i, j, key;
    //第二个元素开始，左边已排序
    for (i = 1; i < n; i++) {
        key = arr[i];   //要插入的元素
        j = i - 1;      //已排序的元素的位置
        //不越界的情况下比较当前元素和要插入的元素（若大则后移）
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;                    //向前比较
        }
        //插入
        arr[j + 1] = key;
    }
}
```

## 快速排序
```c
// 时间复杂度：平均 O(nlogn)
// 快速排序核心：分区函数
int partition(int arr[], int low, int high) {
    int pivot = arr[low];  // 选第一个元素作为基准
    while (low < high) {
        // 从右往左找比基准小的
        while (low < high && arr[high] >= pivot)
            high--;
        arr[low] = arr[high];
        
        // 从左往右找比基准大的
        while (low < high && arr[low] <= pivot)
            low++;
        arr[high] = arr[low];
    }
    arr[low] = pivot;  // 基准归位
    return low;
}

// 快速排序递归函数
void quickSort(int arr[], int low, int high) {
    if (low < high) {
        int pos = partition(arr, low, high); // 找到基准位置
        quickSort(arr, low, pos - 1);        // 左区间递归
        quickSort(arr, pos + 1, high);       // 右区间递归
    }
}
```

## 归并排序
```c
// 时间复杂度 O(nlogn)
// 合并两个有序数组
void merge(int arr[], int left, int mid, int right) {
    int temp[right - left + 1]; // 临时数组
    int i = left, j = mid + 1, k = 0;
    
    // 两个有序序列合并
    while (i <= mid && j <= right) {
        if (arr[i] <= arr[j])
            temp[k++] = arr[i++];
        else
            temp[k++] = arr[j++];
    }
    
    // 剩余元素直接复制
    while (i <= mid) temp[k++] = arr[i++];
    while (j <= right) temp[k++] = arr[j++];
    
    // 拷贝回原数组
    for (i = left, k = 0; i <= right; i++, k++)
        arr[i] = temp[k];
}

// 归并排序递归
void mergeSort(int arr[], int left, int right) {
    if (left < right) {
        int mid = (left + right) / 2;
        mergeSort(arr, left, mid);     // 左半
        mergeSort(arr, mid + 1, right);// 右半
        merge(arr, left, mid, right);  // 合并
    }
}
```

## 堆排序

```c
// 时间复杂度 O(nlogn)
// 调整堆：让以 i 为根的子树成为大顶堆
void heapAdjust(int arr[], int i, int n) {
    int temp = arr[i]; // 根节点
    // 左孩子 2*i+1
    for (int j = 2 * i + 1; j < n; j = 2 * j + 1) {
        // 选左右孩子中较大的
        if (j + 1 < n && arr[j] < arr[j + 1])
            j++;
        // 孩子比根大，孩子上浮
        if (arr[j] > temp) {
            arr[i] = arr[j];
            i = j;
        } else {
            break;
        }
    }
    arr[i] = temp; // 根下沉
}

// 堆排序
void heapSort(int arr[], int n) {
    // 1. 构建大顶堆
    for (int i = n / 2 - 1; i >= 0; i--)
        heapAdjust(arr, i, n);
    
    // 2. 逐步交换堆顶与末尾，调整堆
    for (int i = n - 1; i > 0; i--) {
        // 交换最大值到末尾
        int temp = arr[0];
        arr[0] = arr[i];
        arr[i] = temp;
        // 调整剩余元素
        heapAdjust(arr, 0, i);
    }
}
```

# 查找算法

## 顺序查找

```c
// 适用有序无序
// 顺序查找，返回下标，没找到返回 -1
int seqSearch(int arr[], int n, int key) {
    for (int i = 0; i < n; i++) {
        if (arr[i] == key)
            return i;
    }
    return -1;
}
```

## 二分查找(要求数组有序)

```c
// 二分查找（非递归版，面试最稳）
int binarySearch(int arr[], int n, int key) {
    int left = 0, right = n - 1;

    while (left <= right) {
        int mid = left + (right - left) / 2; // 防溢出

        if (arr[mid] == key)
            return mid;
        else if (arr[mid] < key)
            left = mid + 1;
        else
            right = mid - 1;
    }
    return -1;
}
```
# 概念类

1. 时间复杂度
   - 冒泡 / 选择 / 插入：O(n²)
   - 快排 / 归并 / 堆排：O(nlogn)
   - 二分查找：O(logn)
   - 稳定排序：冒泡、插入、归并
   - 不稳定：快排、选择、堆排
2. 线性表
   - 数组：随机访问 O(1)，插入删除 O(n)
   - 链表：插入删除 O(1)，访问 O(n)
   - 单链表：只能后继
   - 双链表：可前驱可后继
   - 顺序表（数组）优点：查找快
   - 链表优点：插入删除快、不用连续内存
3. 栈 & 队列
   - 栈：后进先出 LIFO
   - 应用：括号匹配、函数调用、递归转非递归
   - 队列：先进先出 FIFO
   - 应用：层序遍历、BFS、缓冲区
4. 树
   - 二叉树第 i 层最多：2^(i-1)
   - 高度 h 的满二叉树节点：2^h − 1
   - n 个节点完全二叉树高度：⌊log₂n⌋ + 1
   - 完全二叉树：除最后一层都满，且靠左
   - 满二叉树：每一层都满
   - 二叉搜索树BST：左 < 根 < 右，中序遍历递增
   - 平衡二叉树AVL：左右子树高度差 ≤1
   - 堆：完全二叉树，大顶堆：父 ≥ 子；小顶堆：父 ≤ 子
   - 红黑树，一种自平衡的二叉搜索树，C++ STL 的 map /set 底层就是它
5. 遍历 & 推导（高频坑点）
   - 前序：根左右
   - 中序：左根右
   - 后序：左右根
   - 已知 前序 + 中序 → 可唯一确定二叉树
   - 已知 后序 + 中序 → 可唯一确定二叉树
   - 前序 + 后序 → 不能唯一确定
6. 图（只记概念）
   - 最短路径：Dijkstra
   - 最小生成树：Kruskal、Prim
7. 查找
   - 顺序查找：无序 / 有序都可，O (n)
   - 二分查找：必须有序，O(log n)
8. 算法思想（常问）
   - 分治法：快排、归并、二分
