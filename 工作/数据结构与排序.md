#  数据结构

## 链表

> 链表：由一系列节点组成，每个节点包含数据域和指针域（指向下一个节点）

### 链表的分类

|分类|特点|适用场景|
|-|-|-|
|单链表|每个节点只有一个 `next` 指针，尾节点指向 `NULL`|简单数据管理|
|双向链表|每个节点有 `prev` 和 `next` 两个指针|双向遍历的场景（内核任务队列）|
|循环链表|尾节点指向头节点，形成环|环型缓冲区|

### 链表与数组的对比

|对比维度|数组|链表|
|-|-|-|
|存储方式|连续内存|分散内存（通过指针链接）|
|大小|固定（除非动态扩容）|动态，可随时增减|
|随机访问|O(1)|O(n)|
|插入/删除|O(n)（需移动元素）|O(1)（已知位置）|
		
### 链表操作（熟记）

```c
#include <stdio.h>
#include <stdlib.h>

// 单链表节点定义
typedef struct ListNode {
    int val;
    struct ListNode *next;
} ListNode;

// 双链表节点定义
typedef struct DNode {
    int data;
    struct DNode *prev;  // 前驱
    struct DNode *next;  // 后继
} DNode;

// 创建新节点
ListNode* createNode(int val) {
    ListNode *node = (ListNode*)malloc(sizeof(ListNode));
    if (!node) {
        printf("内存分配失败\n");
        exit(1);
    }
    node->val = val;
    node->next = NULL;
    return node;
}

// 插入
// 头插法
ListNode* insertAtHead(ListNode *head, int val) {
    ListNode *newNode = createNode(val);
    newNode->next = head;
    return newNode;
}

// 尾插法
ListNode* insertAtTail(ListNode *head, int val) {
    ListNode *newNode = createNode(val);
    if (head == NULL) {
        return newNode;
    }
    ListNode *cur = head;
    while (cur->next) {
        cur = cur->next;    //遍历链表到最后一个节点
    }
    cur->next = newNode;
    return head;
}

// 在指定位置（从0开始）插入，若位置非法则返回原链表
ListNode* insertAtPos(ListNode *head, int val, int pos) {
    if (pos < 0) return head;
    ListNode *newNode = createNode(val);
    if (pos == 0) {
        newNode->next = head;
        return newNode;
    }
    ListNode *cur = head;
    int i = 0;
    while (cur && i < pos - 1) {
        cur = cur->next;        //遍历到插入位置的前一个节点
        i++;
    }
    if (cur == NULL) {  // 位置超出链表长度
        free(newNode);
        return head;
    }
    newNode->next = cur->next;  //新节点指向原位置节点
    cur->next = newNode;        //前驱节点指向新节点
    return head;
}

// 计算链表长度
int getLength(ListNode *head) {
    int len = 0;
    ListNode *cur = head;
    while (cur) {
        len++;
        cur = cur->next;
    }
    return len;
}

//  遍历打印链表
void printList(ListNode *head) {
    if (head == NULL) {
        printf("链表为空\n");
        return;
    }
    ListNode *cur = head;
    while (cur) {
        printf("%d -> ", cur->val);
        cur = cur->next;
    }
    printf("NULL\n");
}

//  删除指定位置的节点（从0开始）
ListNode* deleteAtPos(ListNode *head, int pos) {
    if (head == NULL || pos < 0) return head;

    ListNode *temp = head;
    // 删除头节点
    if (pos == 0) {
        head = head->next;
        free(temp);
        return head;
    }

    // 找到待删除节点的前一个节点
    ListNode *cur = head;
    int i = 0;
    while (cur && i < pos - 1) {
        cur = cur->next;
        i++;
    }
    // 位置越界
    if (cur == NULL || cur->next == NULL) {
        printf("删除位置非法\n");
        return head;
    }

    // 删除节点
    temp = cur->next;
    cur->next = temp->next;
    free(temp);
    return head;
}

// 按值查找节点，返回第一个匹配的节点地址
ListNode* searchByVal(ListNode *head, int target) {
    ListNode *cur = head;
    while (cur) {
        if (cur->val == target) {
            return cur;
        }
        cur = cur->next;
    }
    return NULL; // 未找到
}

// 修改指定位置节点的值
int modifyAtPos(ListNode *head, int pos, int newVal) {
    if (head == NULL || pos < 0) return -1; // 修改失败
    ListNode *cur = head;
    int i = 0;
    while (cur && i < pos) {
        cur = cur->next;
        i++;
    }
    if (cur == NULL) return -1;
    cur->val = newVal;
    return 0; // 修改成功
}

// 销毁整个链表（释放所有内存）
void destroyList(ListNode **head) {
    ListNode *cur = *head;
    ListNode *temp = NULL;
    while (cur) {
        temp = cur;
        cur = cur->next;
        free(temp);
    }
    *head = NULL; // 头指针置空
}
// 反转链表
ListNode* reverseList(ListNode* head) {
    ListNode *prev = NULL, *cur = head, *next;
    while(cur) {
        next = cur->next;
        cur->next = prev;
        prev = cur;
        cur = next;
    }
    return prev;
}

// 判断链表是否有环
int hasCycle(ListNode *head) {
    ListNode *slow = head, *fast = head;
    while(fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
        if(slow == fast) return 1; // 有环
    }
    return 0; // 无环
}

// 找中间节点
ListNode* middleNode(ListNode* head) {
    ListNode *slow = head, *fast = head;
    while(fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
    }
    return slow;
}

// 合并两个有序链表
ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
    // 建一个虚拟头节点，方便操作
    ListNode dummy;
    ListNode* tail = &dummy;
    dummy.next = NULL;

    // 两个链表都不为空，比较大小，小的先接上去
    while (l1 && l2) {
        if (l1->val < l2->val) {
            tail->next = l1;
            l1 = l1->next;
        } else {
            tail->next = l2;
            l2 = l2->next;
        }
        tail = tail->next;
    }

    // 把剩下的链表直接接上
    if (l1) tail->next = l1;
    if (l2) tail->next = l2;

    return dummy.next;
}

// 删除链表倒数第 n 个节点
ListNode* removeNthFromEnd(ListNode* head, int n) {
    ListNode dummy;
    dummy.next = head;
    ListNode *slow = &dummy, *fast = &dummy;

    // 快指针先走 n 步
    for (int i = 0; i < n; i++) {
        fast = fast->next;
    }

    // 快慢一起走，直到快指针到末尾
    while (fast->next) {
        slow = slow->next;
        fast = fast->next;
    }

    // slow->next 就是要删除的节点
    ListNode* temp = slow->next;
    slow->next = temp->next;
    free(temp);

    return dummy.next;
}

// 删除重复节点（有序链表）
ListNode* deleteDuplicates(ListNode* head) {
    ListNode* cur = head;

    while (cur && cur->next) {
        // 当前节点和下一个节点值相同，删除下一个
        if (cur->val == cur->next->val) {
            ListNode* temp = cur->next;
            cur->next = temp->next;
            free(temp);
        } else {
            // 不重复，指针后移
            cur = cur->next;
        }
    }
    return head;
}

// ===================== 主函数：测试所有功能 =====================
int main() {
    ListNode *head = NULL;  // 链表头节点
    ListNode *l1 = NULL, *l2 = NULL;
    ListNode *midNode = NULL;
    ListNode *findNode = NULL;

    printf("========== 1. 插入测试 ==========\n");
    head = insertAtTail(head, 10);     // 尾插 10
    head = insertAtTail(head, 20);     // 尾插 20
    head = insertAtHead(head, 5);      // 头插 5
    head = insertAtPos(head, 15, 2);   // 位置2插入15
    printf("插入完成：");
    printList(head);
    printf("链表长度：%d\n\n", getLength(head));

    printf("========== 2. 修改节点 ==========\n");
    modifyAtPos(head, 2, 99);
    printf("修改位置2为99：");
    printList(head);
    printf("\n");

    printf("========== 3. 查找节点 ==========\n");
    findNode = searchByVal(head, 20);
    if (findNode) printf("找到值：%d\n", findNode->val);
    else printf("未找到\n");
    printf("\n");

    printf("========== 4. 删除指定位置 ==========\n");
    head = deleteAtPos(head, 1);
    printf("删除位置1后：");
    printList(head);
    printf("\n");

    printf("========== 5. 反转链表 ==========\n");
    head = reverseList(head);
    printf("反转后：");
    printList(head);
    printf("\n");

    printf("========== 6. 找中间节点 ==========\n");
    midNode = middleNode(head);
    printf("中间节点值：%d\n\n", midNode->val);

    printf("========== 7. 判断是否有环 ==========\n");
    printf("是否有环：%s\n\n", hasCycle(head) ? "是" : "否");

    printf("========== 8. 删除倒数第n个节点 ==========\n");
    head = removeNthFromEnd(head, 2);
    printf("删除倒数第2个后：");
    printList(head);
    printf("\n");

    printf("========== 9. 删除重复节点 ==========\n");
    head = insertAtTail(head, 20);
    head = insertAtTail(head, 20);
    printf("添加重复值后：");
    printList(head);
    head = deleteDuplicates(head);
    printf("删除重复后：");
    printList(head);
    printf("\n");

    printf("========== 10. 合并两个有序链表 ==========\n");
    l1 = insertAtTail(l1, 1);
    l1 = insertAtTail(l1, 3);
    l1 = insertAtTail(l1, 5);

    l2 = insertAtTail(l2, 2);
    l2 = insertAtTail(l2, 4);
    l2 = insertAtTail(l2, 6);
    ListNode *mergeHead = mergeTwoLists(l1, l2);
    printf("合并后：");
    printList(mergeHead);
    destroyList(&mergeHead);
    printf("\n");

    // 销毁内存
    destroyList(&head);
    printf("========== 链表已销毁 ==========\n");
    printList(head);

    return 0;
}
```

### 嵌入式场景链表的应用

- 内存池管理：空闲块链表，分配时从链表头取，释放时插回
- 内核任务队列：就绪队列、等待队列（双向链表）
- 消息队列：链表作为消息缓冲区，支持动态长度

# 排序算法

## 冒泡排序
```c
//时间复杂度 O(n^2)
//每一轮将最大的数沉底
void bubbleSort(int arr[], int n) {
    //n个元素需要n-1轮
    for (int i = 0; i < n - 1; i++) {
        //n-i-1：因为每轮结束后，最后i个元素已经有序，无需再比较
        for (int j = 0; j < n - i - 1; j++) {
            // 比较相邻两个元素
            if (arr[j] > arr[j + 1]) {
                //大的往后冒
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}
```
## 选择排序
```c
//时间复杂度 O(n^2)
//每一轮选择最矮的一个学生站到当前队列的最前位置
void selectSort(int arr[], int n) {
    //n个元素需要n-1轮
    for (int i = 0; i < n - 1; i++) {
        //假设当前轮次第一个元素是最小
        int min = i;
        //未排序部分找到真正的最小值
        for (int j = i + 1; j < n; j++) {
            //找到更小的，更新最小值位置
            if (arr[j] < arr[min])
                min = j;
        }
        //找到的最小值与当前轮次第一个元素交换
        int temp = arr[i];
        arr[i] = arr[min];
        arr[min] = temp;
    }
}
```
## 插入排序
```c
//时间复杂度 O(n^2)
//打扑克牌理牌，左边的部分已经排序，右边的部分还没排序，每次摸一张牌，都在正确的位置插入
void insertSort(int arr[], int n) {
    int i, j, key;
    //第二个元素开始，左边已排序
    for (i = 1; i < n; i++) {
        key = arr[i];   //要插入的元素
        j = i - 1;      //已排序的元素的位置
        //不越界的情况下比较当前元素和要插入的元素（若大则后移）
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;                    //向前比较
        }
        //插入
        arr[j + 1] = key;
    }
}
```
